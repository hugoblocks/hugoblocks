```
- kind: benefits
  id:
  title:
  items:

- kind: action-card
  id:
  title:
  body:

    url: or ref:
    target: (_blank, etc)
    name:

- kind: features
  title:
  items:
    - title:
      image:
      tagline:
      copy:

- kind: hero
  image:
  title:
  tagline:
  notes:
    - priority:
      title:
  actions:
    left:
        url:
        name:
    right:
        url:
        name:
```
